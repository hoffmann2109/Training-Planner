7    public class TrainingWeek {
8        private List<Exercise> exercises = new ArrayList<>();
9        private int week;
10       private Map<MuscleGroup, Integer> setsPerWeek = new HashMap<>();
11   
12       public TrainingWeek(int week) {
13           this.week = week;
14       }
15   
16       public void getSetsPerWeek(){
17           System.out.println("-----------------------------");
18           System.out.println("Sets completed in Week "+ week + ": " + "\n") ;
19           for (MuscleGroup muscle: setsPerWeek.keySet()) {
20               System.out.println(muscle + ": " + setsPerWeek.get(muscle) + " sets");
21           }
22           System.out.println("-----------------------------");
23       }
24   
25       public void addExercises(List<List<String>> array) {
26           int length = array.size();
27           for (int i = 0; i < length; i++){
28   
29               List<String> innerList = array.get(i);
30               if (innerList.size() < 6) {
31                   continue;
32               }
33   
34               String name = array.get(i).get(0).trim();
35   
36               String muscleGroupString = array.get(i).get(1).trim().toUpperCase();
37               muscleGroupString = muscleGroupString.replace(' ', '_');
38               MuscleGroup muscleGroup = MuscleGroup.valueOf(muscleGroupString);
39               int setCount = Integer.parseInt(array.get(i).get(2).trim());
40               double weight = Double.parseDouble(array.get(i).get(3).trim());
41               List<Set> sets = new ArrayList<>();
42   
43               for (int j = 0; j < setCount; j++){
44                   // reps:
45                   String sReps = array.get(i).get(4).trim();
46                   String[] repArray = sReps.split(",");
47                   int reps = Integer.parseInt(repArray[j].trim());
48   
49                   //rpe:
50                   String sRpe = array.get(i).get(5).trim();
51                   String[] rpeArray = sRpe.split(",");
52                   double rpes = Double.parseDouble(rpeArray[j].trim());
53   
54                   // add to sets:
55                   sets.add(new Set(weight, reps, rpes));
56               }
57               this.exercises.add(new Exercise(name, muscleGroup, sets));
58               // Accumulate set counts per muscle group
59               this.setsPerWeek.merge(muscleGroup, setCount, Integer::sum);
60   
61           }
62       }
63   
64       public void printExercises() {
65           for (Exercise exercise : exercises) {
66               System.out.println(exercise);
67           }
68       }
